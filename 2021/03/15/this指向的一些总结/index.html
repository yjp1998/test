this指向的一些总结

发表于 2021-03-15

谨记：this 永远指向最后调用它的那个对象
基本
1、用 let 或 const 声明的变量不会绑定到全局对象中
有关隐式绑定的
两种情况：
• 使用另一个变量来给函数取别名
• 将函数作为参数传递时会被隐式赋值，回调函数丢失 this 绑定
如果你把一个函数当成参数传递到另一个函数的时候，也会发生隐式丢失的问题，且与包裹着它的函数的this指向无关。在非严格模式下，会把该函数的this绑定到window上，严格模式下绑定到undefined。
setTimeout 中的函数，也存在隐式绑定的隐式丢失，也就是我们讲函数作为参数传递时会被隐式赋值，回调函数丢失 this 绑定，因此 setTimeou 中的函数内的 this 是指向 window 的。
若给 setTimeout 的回调函数使用 call 或 apply ，就指向绑定的对象
setTimeout(function() {}.call(xxx), 0)
有关显示绑定的
foo.call() 和 foo().call() 的区别，一个是针对于函数，一个针对于函数的返回值
function foo() {
console.log(this.a);
}
var obj = { a: 1 }
var a = 2
foo()
foo.call(obj)
foo().call(obj)
// 输出结果
2
1
2
Uncaught TypeError: Cannot read property ‘call’ of undefined
// 解释：因为foo() 函数的返回值是 undefined，所以报错
修改一下foo 其余不变
function foo() {
console.log(a)
return function() {
console.log(this.a)
}
}
// 输出结果
2
1
2
1
注意，若是foo().bind(obj)，bind() 方法不会立即调用，所以只是给foo() 返回值绑定了 this 指向而没有调用
var obj = {
a: 1,
foo: function(b) {
b = b || this.a
return function(c) {
console.log(this.a + b + c)
}
}
}
var a = 2;
var obj2 = { a: 3 }
obj.foo(a).call(obj2, 1) // 6
obj.foo.call(obj2)(1) // 6
// 解析
1，开始调用obj.foo(a) 将全局变量中的2传入参数并赋值给b
由于闭包的原因，所以返回的函数能访问到b，b = 2
obj.foo(a) 返回值是函数, 调用call()的是返回值，所以this指向obj2，this.a = 3
c作为参数传进为1
所以a + b + c = 3 + 2 + 1 = 6
2,obj.foo.call()是 obj.foo 调用的call(),所以foo 的this 指向obj2
因为没传参数，b = this.a，所以 b = 3
(1)是foo返回的函数调用, c = 1, this 指向 window，所以 this.a = 2,
所以 a + b + c = 2 + 3 + 1 = 6
3、forEach、map、filter 的第二个参数能绑定 this，不传入的情况下指向 window
function foo (item) {
console.log(item, this.a)
}
var obj = {
a: ‘obj’
}
var a = ‘window’
var arr = [1, 2, 3]
// arr.forEach(foo, obj)
// arr.map(foo, obj)
arr.filter(function (i) {
console.log(i, this.a)
return i > 2
}, obj)
输出：
“obj”
“obj”
“obj”
有关 new 绑定的
function Person (name) {
this.name = name
this.foo1 = function () {
console.log(this.name)
}
this.foo2 = function () {
return function () {
console.log(this.name)
}
}
}
var person1 = new Person(‘person1’)
person1.foo1()
person1.foo2()()
输出
person1
空 // window 对象中并不存在 name 属性，所以打印出来的是空
使用 new 函数创建的对象和字面量形式创建出来的对象没什么大区别，如果对象中有属性是函数类型的话，并且不是箭头函数，那么解法都一样。
var name = ‘window’
function Person (name) {
this.name = name
this.foo = function () {
console.log(this.name)
return function () {
console.log(this.name)
}
}
}
var person2 = {
name: ‘person2’,
foo: function() {
console.log(this.name)
return function () {
console.log(this.name)
}
}
}

var person1 = new Person(‘person1’)
person1.foo()()
person2.foo()()
输出
‘person1’
‘window’
‘person2’
‘window’
有关箭头函数
它里面的 this 是由外层作用域来决定的，且指向函数定义时的 this 而并非执行时。
箭头函数中没有 this 绑定，必须通过查找作用域链来决定其值，如果箭头函数被非箭头函数包含，则 this 绑定的是最近一层非箭头函数的 this，否则，this 为 undefined。
var obj = {
name: ‘obj’,
foo1: () => {
console.log(this.name)
},
foo2: function () {
console.log(this.name)
return () => {
console.log(this.name)
}
}
}
var name = ‘window’
obj.foo1()
obj.foo2()()
输出
‘window’ // 外层作用域是 window，对象 obj 不属于作用域
‘obj’ // obj.foo2() 不是箭头函数，this 调用它的 obj 对象
‘obj’ // 返回的是箭头函数，this 由外层作用域决定，也就是函数 foo2，this指向与foo2的this一样，也就是obj
注意：作用域只有全局作用域 window 和局部作用域函数（在没有块级作用域之前），对象字面量的写法并不会产生自己的作用域。
构造函数对象中普通函数与箭头函数区别：
var name = ‘window’
function Person (name) {
this.name = name
this.foo1 = function () {
console.log(this.name)
}
this.foo2 = () => {
console.log(this.name)
}
}
var person2 = {
name: ‘person2’,
foo2: () => {
console.log(this.name)
}
}
var person1 = new Person(‘person1’)
person1.foo1()
person1.foo2()
person2.foo2()
输出
person1
person1
// 外层作用域是函数 Person，且这个是构造函数，并且使用了 new 来生成对象 person1，所以指向person1
window
// person 为对象字面量，不会产生自己的作用域，所以 this 指向 window
记住，this 由外层作用域决定，且指向函数定义时的 this 而非执行时。
箭头函数的 this 无法通过 bind、call、apply 来修改，但可以通过改变作用域中 this 的指向来间接修改。
var name = ‘window’
function Person (name) {
this.name = name
this.obj = {
name: ‘obj’,
foo1: function () {
return function () {
console.log(this.name)
}
},
foo2: function () {
return () => {
console.log(this.name)
}
}
}
}
var person1 = new Person(‘person1’)
var person2 = new Person(‘person2’)
person1.obj.foo1()() // ‘window’
person1.obj.foo1.call(person2)() // ‘window’
person1.obj.foo1().call(person2) // ‘person2’
person1.obj.foo2()() // ‘obj’
// 箭头函数的 this 由第一层普通函数决定，第一层函数最后由obj调用
person1.obj.foo2.call(person2)() // ‘person2’
person1.obj.foo2().call(person2) // ‘obj’
this 永远指向最后调用它的那个对象。
function foo() {
console.log( this.a );
}
var a = 2;
(function(){
“use strict”;
foo();
})();
输出
2
// 使用了”use strict”开启严格模式会使得”use strict”以下代码的this为undefined，
// 也就是这里的立即执行函数中的this是undefined。
// 但是调用foo()函数的依然是window，所以foo()中的this依旧是window，所以会打印出2。
// 如果使用 this.foo() 调用的话就报错
